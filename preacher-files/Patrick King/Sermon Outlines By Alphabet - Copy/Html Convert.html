# Path to your Sermon Outlines By Alphabet folder

$root = "C:\Users\patri\OneDrive\Documents\Sermon Outlines\Sermon Outlines By Alphabet"

# Load Word COM object

$word = New-Object -ComObject Word.Application

$word.Visible = $false

# Get all DOCX files in all subfolders (A–Z)

$files = Get-ChildItem -Path $root -Recurse -Filter *.docx

foreach ($file in $files) {

try {

Write-Host "Converting: $($file.FullName)"

# Open the Word document

$doc = $word.Documents.Open($file.FullName)

# Build output HTML filename

$htmlPath = [System.IO.Path]::ChangeExtension($file.FullName, ".html")

# Save as Filtered HTML (clean for websites)

$doc.SaveAs([ref] $htmlPath, [ref] 10) # 10 = wdFormatFilteredHTML

# Close the document

$doc.Close()

}

catch {

Write-Host "Error converting $($file.FullName): $_"

}

}

# Quit Word

$word.Quit()

Write-Host "Conversion complete!"
